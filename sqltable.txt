CREATE TABLE orders (
    id INT AUTO_INCREMENT PRIMARY KEY,
    order_date DATE,
    order_number VARCHAR(30),
    department VARCHAR(100),
    employee VARCHAR(100),
    payment_method VARCHAR(100),
    customer_name VARCHAR(150),
    phone_number VARCHAR(50),
    personal_id VARCHAR(30),
    product_name VARCHAR(255),
    quantity VARCHAR(20),
    revenue VARCHAR(50),
    contribution VARCHAR(100),
    bank VARCHAR(100),
    city VARCHAR(100),
    location_details TEXT,
    item_carrier VARCHAR(100),
    delivery_type VARCHAR(100),
    planned_delivery_time VARCHAR(50),
    order_ready_status VARCHAR(100),
    item_collection_note TEXT,
    location_movement TEXT,
    order_status_1 VARCHAR(100),
    tracking_code TEXT,
    delivery_status_2 VARCHAR(100),
    delivery_date_fx DATE,
    days_diff_fx VARCHAR(20),
    order_deadline_fx VARCHAR(50),
    standard_deadline DATE,
    status_update TEXT,
    failed_delivery_comment TEXT,
    resend_date DATE
);
